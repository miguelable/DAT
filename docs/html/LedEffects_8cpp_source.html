123void fadeInOut(uint16_t time, RgbColor target, uint16_t wait, uint16_t repeat)
124{
125 uint16_t currentRepeat = repeat; // Mantener un contador local del número de repeticiones

127 // Comprobar si se envió un nuevo efecto durante el bucle
128 LedEffect newEffect;

133 // Ejecutar las animaciones de fade in y fade out
134 fadeIn(time, target);

136 // Esperar antes de la próxima iteración
137 vTaskDelay(wait / portTICK_PERIOD_MS);
138 // Reducir el contador solo si repeat es mayor a 0

140 currentRepeat--;
141 }
142
143 } while (repeat == 0 || currentRepeat > 0); // Repetir indefinidamente si repeat == 0

222void ledTaskFunction(void* pvParameters)
223{
224 LedEffect currentEffect = None;
225 while (true) {
226 // Esperar un nuevo efecto en la cola
227 if (xQueueReceive(effectQueue, &currentEffect, portMAX_DELAY)) {
228 // Si se recibió un nuevo efecto, procesarlo
229 if (currentEffect != None) {
230 // Limpia la cola para evitar que se acumulen efectos antiguos
231 xQueueReset(effectQueue);
232 // Ejecutar el efecto
