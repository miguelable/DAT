109void HCSR04::sensorDataTask(void* pvParameters)
110{
111 HCSR04* sensor = static_cast<HCSR04*>(pvParameters);
112 while (true) {
113 digitalWrite(sensor->_trigPin, LOW);
114 delayMicroseconds(2);
115 digitalWrite(sensor->_trigPin, HIGH);
116 delayMicroseconds(10);
117 digitalWrite(sensor->_trigPin, LOW);
118
119 sensor->_duration = pulseIn(sensor->_echoPin, HIGH);
120 sensor->_distance = (sensor->_duration * .0343) / 2;
121
122 if (sensor->_distance < sensor->_threshold && sensor->_callback != NULL)
123 if (sensor->_distance > 5) {
124 // Serial.printf("Distance: %0.2f cm\n", sensor->_distance);
125 sensor->_callback();
126 }
127 vTaskDelay(20 / portTICK_PERIOD_MS);
128 }
129}
