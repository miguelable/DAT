115 DynamicJsonDocument doc(200);

135 vTaskDelay(4000);

139 log_i("Acceso permitido a la atraccion");
140 setLedEffect(Success);

142 vTaskDelay(4000);

144 }

146 log_e("Mensaje no valido");
147 }
148 }

150 log_e("Mensaje no valido");
