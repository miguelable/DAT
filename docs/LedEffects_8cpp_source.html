130void fadeInOut(uint16_t time, RgbColor target, uint16_t wait, uint16_t repeat)
131{
132 uint16_t currentRepeat = repeat; // Mantener un contador local del número de repeticiones
133 do {
134 // Comprobar si se envió un nuevo efecto durante el bucle
135 LedEffect newEffect;
136 if (xQueuePeek(effectQueue, &newEffect, 0) == pdPASS && newEffect != None) {
137 // Si hay un nuevo efecto en la cola, salir del bucle
138 break;
139 }
140 // Ejecutar las animaciones de fade in y fade out
141 fadeIn(time, target);
142 fadeOut(time);
143 // Esperar antes de la próxima iteración
144 vTaskDelay(wait / portTICK_PERIOD_MS);
145 // Reducir el contador solo si repeat es mayor a 0
146 if (repeat > 0) {
147 currentRepeat--;
148 }
149
150 } while (repeat == 0 || currentRepeat > 0); // Repetir indefinidamente si repeat == 0
151}

178void handleEffect(LedEffect effect)
179{
180 switch (effect) {
181 case Clear:
182 fadeOut(500);
183 break;
184 case Flashing:
185 fadeIn(500, RgbColor(0, 255, 255));
186 break;
187 case Processing:
188 fadeInOut(1000, RgbColor(0, 255, 255), 200, 0);
189 break;

192 vTaskDelay(3000 / portTICK_PERIOD_MS);
193 fadeOut(500);
194 setLedEffect(Waiting);
195 break;

197 fadeIn(500, RgbColor(255, 0, 0));
198 vTaskDelay(3000 / portTICK_PERIOD_MS);

220// Tarea principal de gestión de LED
231void ledTaskFunction(void* pvParameters)
232{
233 LedEffect currentEffect = None;
234 while (true) {
235 // Esperar un nuevo efecto en la cola
236 if (xQueueReceive(effectQueue, &currentEffect, portMAX_DELAY)) {
237 // Si se recibió un nuevo efecto, procesarlo
238 if (currentEffect != None) {
239 // Limpia la cola para evitar que se acumulen efectos antiguos
240 xQueueReset(effectQueue);
241 // Ejecutar el efecto
242 handleEffect(currentEffect);
243 }
244 }
245 }
246}
